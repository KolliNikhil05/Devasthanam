using DevasthanamBAL;
using System;
using System.Data;

namespace Devasthanam.Views
{
    public partial class test : System.Web.UI.Page
    {

        testBAL obj = new testBAL();
        protected void Page_Load(object sender, EventArgs e)
        {
            if (!IsPostBack)
            {
                // Retrieve skills using the BAL
                DataTable skillsTable = obj.GetSkills();

                // Bind skills to the CheckBoxList
                CheckBoxList1.DataSource = skillsTable;
                CheckBoxList1.DataTextField = "Skill"; // Column name for skill name
                CheckBoxList1.DataValueField = "SkillId"; // Column name for skill ID
                CheckBoxList1.DataBind();
            }
        }
    }
}

using System.Configuration;
using System.Data;
using System.Data.SqlClient;
namespace DevasthanamDAL
{
    public class testDAL
    {
        public string connectionString = ConfigurationManager.ConnectionStrings["connection_data"].ConnectionString;
        public DataTable GetSkills()
        {
            using (SqlConnection connection = new SqlConnection(connectionString))
            {
                using (SqlCommand cmd = new SqlCommand("Skills_Get", connection))
                {
                    cmd.CommandType = CommandType.StoredProcedure;
                    connection.Open();

                    DataTable skillsDataTable = new DataTable();
                    using (SqlDataAdapter adapter = new SqlDataAdapter(cmd))
                    {
                        adapter.Fill(skillsDataTable);
                    }

                    return skillsDataTable;
                }
            }
        }
    }

}
