@SurName varchar(15),      
@Phone varchar(10),      
@Email varchar(25),      
@Gender varchar(20),      
@DOB varchar(8),      
@City varchar(20),      
@UAddress varchar(150),    
@Qualification varchar(15),  
@UPercentage varchar(6),  
@UCertificate varchar(200)  
)      
as      
begin      
insert into registration     
(  
[uName],    
[SurName],   
[Phone],   
[Email],   
[Gender] ,   
[DOB],    
[City],    
[UAddress],   
[Qualification],   
[UPercentage],    
[UCertificate],      
[CreatedBy],    
[CreatedDate],    
[IsActive]    
)      
values  
(  
@uName,      
@SurName,      
@Phone,      
@Email,      
@Gender,      
@DOB,      
@City,      
@UAddress,    
@Qualification,  
@UPercentage,  
@UCertificate,  
@uName,  
getdate(),  
'y'  
)      
end

CREATE PROCEDURE InsertSignupData  
    @userId VARCHAR(10),  
    @Password VARCHAR(15),  
    @cPassword VARCHAR(15)  
AS  
BEGIN  
    SET NOCOUNT ON;  
  
       
    IF EXISTS (SELECT 1 FROM signup WHERE userId = @userId)  
    BEGIN  
          
        SELECT 'Already Registered' AS MSG, 0 AS FLAG;  
    END  
    ELSE  
    BEGIN  
           
        INSERT INTO signup (userId, [Password], cPassword, createdBy, createdTime, isActive, [role])  
        VALUES (@userId, @Password, @cPassword, @userId, GETDATE(), 'y', 'user');  
  
          
        SELECT 'INSERTED SUCCESSFULLY' AS MSG, 1 AS FLAG;  
    END;  
END;  
  
select*from signup
